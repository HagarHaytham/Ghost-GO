def main():
    board_size = 19
    game = goboard.GameState.new_game(board_size)

        gotypes.Player.black: 0,
        gotypes.Player.white: 0,
    }

        game , captures = monte_carlo_tree_search( game,player,num_rounds,captures,depth)
